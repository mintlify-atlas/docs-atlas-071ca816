---
title: Constructor Options
description: Configuration options for the Editor constructor (IEditorOption)
---

The `IEditorOption` interface defines all available configuration options when creating an Editor instance.

## Editor Mode

<ParamField path="mode" type="EditorMode" default="EditorMode.EDIT">
  Editor operating mode:
  - `EDIT` - Full editing capabilities (default)
  - `CLEAN` - Hide auxiliary elements
  - `READONLY` - View-only mode
  - `FORM` - Only controls are editable
  - `PRINT` - Print preview mode
  - `DESIGN` - Design mode (ignores readonly/deletable rules)
  - `GRAFFITI` - Drawing/annotation mode
</ParamField>

## Locale

<ParamField path="locale" type="string" default="'en'">
  Language/locale code for internationalization (e.g., 'en', 'zh-CN')
</ParamField>

## Default Styles

<ParamField path="defaultType" type="string" default="'TEXT'">
  Default element type for new content
</ParamField>

<ParamField path="defaultColor" type="string" default="'#000000'">
  Default text color (hex format)
</ParamField>

<ParamField path="defaultFont" type="string" default="'Microsoft YaHei'">
  Default font family
</ParamField>

<ParamField path="defaultSize" type="number" default="16">
  Default font size in pixels
</ParamField>

<ParamField path="minSize" type="number" default="5">
  Minimum allowed font size
</ParamField>

<ParamField path="maxSize" type="number" default="72">
  Maximum allowed font size
</ParamField>

## Layout & Dimensions

<ParamField path="width" type="number" default="794">
  Page width in pixels (A4: 794px at 96dpi)
</ParamField>

<ParamField path="height" type="number" default="1123">
  Page height in pixels (A4: 1123px at 96dpi)
</ParamField>

<ParamField path="scale" type="number" default="1">
  Zoom scale factor (1 = 100%)
</ParamField>

<ParamField path="pageGap" type="number" default="20">
  Gap between pages in pixels
</ParamField>

<ParamField path="margins" type="IMargin">
  Page margins:
  ```typescript
  {
    top: number    // default: 100
    right: number  // default: 120
    bottom: number // default: 100
    left: number   // default: 120
  }
  ```
</ParamField>

<ParamField path="pageMode" type="PageMode" default="PageMode.PAGING">
  Page layout mode:
  - `PAGING` - Multi-page with page breaks
  - `CONTINUITY` - Continuous scroll (no pages)
</ParamField>

<ParamField path="paperDirection" type="PaperDirection" default="PaperDirection.VERTICAL">
  Paper orientation:
  - `VERTICAL` - Portrait
  - `HORIZONTAL` - Landscape
</ParamField>

## Text & Spacing

<ParamField path="defaultBasicRowMarginHeight" type="number" default="8">
  Base row margin height
</ParamField>

<ParamField path="defaultRowMargin" type="number" default="0">
  Default row margin
</ParamField>

<ParamField path="defaultTabWidth" type="number" default="32">
  Width of tab character in pixels
</ParamField>

<ParamField path="wordBreak" type="WordBreak" default="WordBreak.BREAK_WORD">
  Word breaking behavior:
  - `BREAK_WORD` - Break at word boundaries
  - `BREAK_ALL` - Break anywhere
</ParamField>

## Example

```typescript
import Editor, { EditorMode, PageMode } from '@hufe921/canvas-editor'

const options: IEditorOption = {
  mode: EditorMode.EDIT,
  defaultFont: 'Arial',
  defaultSize: 16,
  defaultColor: '#333333',
  width: 794,
  height: 1123,
  margins: {
    top: 100,
    right: 120,
    bottom: 100,
    left: 120
  },
  pageMode: PageMode.PAGING,
  rangeColor: '#AECBFA',
  rangeAlpha: 0.6
}

const editor = new Editor(container, data, options)
```
